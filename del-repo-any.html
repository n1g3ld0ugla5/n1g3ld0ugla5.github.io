Rules:

  
- rule: Delete Repository
  


    desc: Detect the deletion of a repository
  


    condition: github.type=repository and github.action=deleted  
  


    output: A repository was deleted (repository=%github.repo
     repo_owner=%github.owner org=%github.org user=%github.user)  
  

  
    priority: INFO
  

 
    source: github
